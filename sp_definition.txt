 and [Lower] <= @Age and LabTestAge.LabTest = @LabTestID and LabTestAge.IsDeleted=0 order by [Lower] asc

									case when ISNULL(DefaultMin, '') <> '' and ISNULL(DefaultMax, '') <> '' then  cast(Cast(DefaultMin as float) as nvarchar(25)) + ' - ' + cast(Cast(DefaultMax as float) as nvarchar(25))

									Set @RefRange = @Min + ' - ' + @Max
                                                                                                                                                                                                                 
								else if (ISNULL(@Min, '') <> '')
                                                                                                                                                                                                                     
									Set @RefRange = '> ' + @Min
                                                                                                                                                                                                                         
								else if (ISNULL(@Max, '') <> '')

							Begin
                                                                                                                                                                                                                                                 
								Select @RefRange = 
                                                                                                                                                                                                                                  
								(
                                                                                                                                                                                                                                                    
									select distinct  STUFF((select N'<BR/>' + Description + ': ' +
                                                                                                                                                                                      
									case when FemaleNormalMin is not null and FemaleNormalMax is not null then  cast(Cast(FemaleNormalMin as float) as nvarchar(25)) + ' - ' + cast(Cast(FemaleNormalMax as float) as nvarchar(25))

], IsCollection, ErrorMin, ErrorMax, GUID, Ref_Range, Prefix, Suffix, LowPanicIndex, HighPanicIndex, CreatedBy)

rorMax, @GUID, @RefRange, @Prefix, @Suffix, @PanicMin, @PanicMax, @Session)
